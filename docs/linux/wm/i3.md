

```bash
$mode+Enter              # open terminal
$mode+d                  # start menu
$mode+Shift+q            # close windows
$mode_j/k/l/;            # navigate windows
$mode-v                  # tile vertical
$mode-h                  # tile horizontally
$mode+num                # switch workspace
$mode+Ctrl+jh8           # move window to workspace
$mode+Space              # focus tile/float windows
$mode (hold)             # move float window with mouse
$mode+r                  # resize container with j/k/l/; (Esc/Enter to exit)
$mode+Shift+g            # resize gap o/i (outer/inner) and +/-/0
$mode+e                  # tile mode (toggle split)
$mode+s                  # stack mode
$mode+w                  # tab mode
$mode+u/y/n              # window borders: none/2px/normal
$mode+Shift+c            # reload configuration
$mode+Shift+r            # restart in place
$mode+9                  # lock screen
$mode+0                  # exit session (l)ock, (e)xit, (s)uspend, (r)eboot,
                         #              switch_(u)ser, (h)ibernate, 
                         #              (Shift+s)hutdown or Esc/Enter
$mode+Shift+r            # logout from Xsession
```
